name: Startup

on:
  push:
    branches:
      - main
      - master
  pull_request:
    branches:
      - '*'
  workflow_dispatch:

jobs:
  startup:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4

          node-version: '20'

      - name: Install dependencies
        run: npm ci
